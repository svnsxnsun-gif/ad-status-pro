export default async function handler(req, res) {
  if (req.method !== 'POST') return res.status(405).send('Method Not Allowed');

  const { domain, industry, keywords } = req.body;
  const API_KEY = process.env.GEMINI_API_KEY; // Vercel 설정창에 넣을 키 이름

  try {
    const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${API_KEY}`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        contents: [{
          parts: [{
            text: `Professional marketing analysis for Domain(${domain}), Industry(${industry}), Keywords(${keywords}). Summarize in Korean with keywords and bullet points.`
          }]
        }]
      })
    });

    const data = await response.json();
    const text = data.candidates[0].content.parts[0].text;
    res.status(200).json({ text });
  } catch (error) {
    res.status(500).json({ error: 'Internal Server Error' });
  }
}
